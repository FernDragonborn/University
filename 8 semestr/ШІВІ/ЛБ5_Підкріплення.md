#### 3.1. Завдання 1: ε-greedy k-armed bandit

##### 3.1.1. Ініціалізація параметрів

```python
import pygame-ce as pygame
import sys
import numpy as np
from pygame.locals import *

# Параметри варіанту 2
K = 4
d = [
    np.random.uniform(-5.0, 18.0, 250),
    np.random.uniform(-8.0, 10.0, 250),
    np.random.uniform(0.0, 20.0, 250),
    np.random.uniform(-5.0, 36.0, 250)
]
eps = 0.5
max_iter = 200

# Ініціалізація
q_a = [0.0] * K       # оцінки цінності дій
n_a = [0] * K          # кількість вибору кожної дії
score = 0
i = 0
```

Маємо 4 руки (arms) з різними розподілами винагород. Четверта рука має найширший діапазон (-5, 36) і потенційно найвищу середню винагороду.

```python
def draw_bar(surface, x, y, w, h, color, label, value):
    """Малює стовпчик з підписом."""
    bar_h = max(0, int(h * value / 20))
    pygame.draw.rect(surface, color, (x, y + h - bar_h, w, bar_h))
    pygame.draw.rect(surface, TEXT_COLOR, (x, y, w, h), 1)
    lbl = font.render(label, True, TEXT_COLOR)
    surface.blit(lbl, (x + w // 2 - lbl.get_width() // 2, y + h + 5))
    val = font.render(f'{value:.2f}', True, TEXT_COLOR)
    surface.blit(val, (x + w // 2 - val.get_width() // 2, y - 20))

def draw_info(surface, iteration, score, action_type, chosen):
    """Виводить інформацію про поточну ітерацію."""
    texts = [
        f'Iteration: {iteration}/{max_iter}',
        f'Total Score: {score:.2f}',
        f'Action: {action_type} (arm {chosen})',
        f'ε = {eps}'
    ]
    for j, t in enumerate(texts):
        txt = font.render(t, True, TEXT_COLOR)
        surface.blit(txt, (550, 30 + j * 25))
```

##### 3.1.4. Ігровий цикл

```python
running = True
clock = pygame.time.Clock()

while running:
    for event in pygame.event.get():
        if event.type == QUIT:
            running = False

    if i < max_iter:
        # ε-greedy вибір дії
        if np.random.random() < eps:
            action = np.random.randint(0, K)
            action_type = 'exploration'
        else:
            action = np.argmax(q_a)
            action_type = 'exploitation'

        # Отримання винагороди
        reward = d[action][i % len(d[action])]
        score += reward

        # Оновлення оцінки цінності
        n_a[action] += 1
        q_a[action] += (reward - q_a[action]) / n_a[action]

        i += 1
    else:
        action_type = 'done'
        action = -1

    # Відмалювання
    screen.fill(BG_COLOR)

    title = big_font.render('K-Armed Bandit (ε-greedy)', True, TEXT_COLOR)
    screen.blit(title, (20, 10))

    # Стовпчики Q-значень
    bar_w = 80
    bar_h = 300
    for k in range(K):
        x = 50 + k * (bar_w + 30)
        draw_bar(screen, x, 80, bar_w, bar_h, COLORS[k],
                 f'Arm {k}', q_a[k])
        cnt = font.render(f'n={n_a[k]}', True, TEXT_COLOR)
        screen.blit(cnt, (x + bar_w // 2 - cnt.get_width() // 2,
                          80 + bar_h + 25))

    draw_info(screen, i, score, action_type, action)

    pygame.display.flip()
    clock.tick(30)

pygame.quit()
```

#### 3.2. Завдання 2: Gridland (Value Iteration)

##### 3.2.1. Параметри задачі

```python
import numpy as np
import pygame
from pygame.locals import *

# Параметри
GRID_SIZE = 5
gamma = 0.9
threshold = 0.001

# Спеціальні клітинки (варіант 2)
# A: (1, 2) → A': (3, 1), винагорода +7
A_pos = (1, 2)
A_prime = (3, 1)
A_reward = 7

# Дії: вверх, вниз, вліво, вправо
actions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
action_names = ['↑', '↓', '←', '→']
```

##### 3.2.2. Value Iteration

```python
V = np.zeros((GRID_SIZE, GRID_SIZE))

iteration = 0
while True:
    delta = 0
    V_new = np.copy(V)

    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            # Спеціальна клітинка A
            if (r, c) == A_pos:
                V_new[r, c] = A_reward + gamma * V[A_prime]
            else:
                # Звичайна клітинка — max по діях
                best_value = float('-inf')
                for dr, dc in actions:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                        # Всередині поля: reward = 0
                        value = 0 + gamma * V[nr, nc]
                    else:
                        # За межами: reward = -1, залишаємось
                        value = -1 + gamma * V[r, c]
                    best_value = max(best_value, value)

                V_new[r, c] = best_value

            delta = max(delta, abs(V_new[r, c] - V[r, c]))

    V = V_new
    iteration += 1

    if delta < threshold:
        break

print(f'Збіглося за {iteration} ітерацій')
print('Таблиця цінності станів v*:')
print(np.round(V, 2))
```

##### 3.2.3. Визначення оптимальної політики

```python
policy = np.empty((GRID_SIZE, GRID_SIZE), dtype='U2')

for r in range(GRID_SIZE):
    for c in range(GRID_SIZE):
        if (r, c) == A_pos:
            policy[r, c] = '★'
            continue

        best_action = 0
        best_value = float('-inf')
        for a_idx, (dr, dc) in enumerate(actions):
            nr, nc = r + dr, c + dc
            if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
                value = 0 + gamma * V[nr, nc]
            else:
                value = -1 + gamma * V[r, c]

            if value > best_value:
                best_value = value
                best_action = a_idx

        policy[r, c] = action_names[best_action]

print('Оптимальна політика π*:')
print(policy)
```

Всі стрілки вказують у напрямку клітинки A (1,2), оскільки вона має найвищу цінність (20.35) завдяки постійній винагороді +7 з телепортацією. Верхній рядок рухається вниз до A, решта — вгору.

### 4. Висновки

У ході виконання лабораторної роботи вивчено основи навчання з підкріпленням:

1. **ε-greedy політика** для k-armed bandit реалізує компроміс між дослідженням (exploration) та використанням (exploitation). При ε=0.5 агент витрачає половину часу на дослідження нових дій. Формула оновлення оцінки: Q(a) ← Q(a) + (1/n)(R − Q(a)).

2. **Value Iteration** для задачі Gridland обчислює оптимальну функцію цінності v* шляхом ітеративного застосування рівняння Белмана: v(s) = max_a [R(s,a) + γ·v(s')]. Алгоритм збігається за 86 ітерацій при γ=0.9 та порозі 0.001.

3. Оптимальна політика π* спрямовує агента до клітинки A з винагородою +7. Всі стрілки вказують у напрямку найвищої цінності.

4. Параметр **ε** впливає на баланс exploration/exploitation: менше ε → більше exploitation → вища сумарна винагорода за умови достатньої кількості кроків для оцінки дій. Параметр **γ** (discount factor) визначає, наскільки агент цінує майбутні винагороди.
